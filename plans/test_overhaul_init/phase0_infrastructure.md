```toml
    "pragma: no cover",

    "@abstractmethod",
```

Key decisions:
- `branch = true` — we target 100% branch coverage
- `omit` the compat layer (import shims) and log module (trivial)
- Exclude `@overload` lines (type stubs, never executed)
- Exclude `@abstractmethod` lines (implemented in subclasses)

### 4b: Create ratcheting script

**File:** `ci/check_coverage.py`

```python
#!/usr/bin/env python3
"""Coverage ratcheting script — validates coverage against thresholds.

Reads an existing .coverage.json report (generated by `make test-cov`)
and checks per-module branch coverage against recorded thresholds.

This script does NOT run tests — it only validates an existing report.
Run `make test-cov` first to generate .coverage.json.

Usage:
    python ci/check_coverage.py                    # Check mode
    python ci/check_coverage.py --update           # Update thresholds
    python ci/check_coverage.py --verbose          # Show per-file breakdown
    python ci/check_coverage.py --report FILE      # Use specific report file

Exit codes:
    0 — All modules at or above threshold
    1 — One or more modules below threshold
    2 — Missing report file or threshold file
"""
from __future__ import annotations

import json
import sys
from pathlib import Path

THRESHOLD_FILE = Path(__file__).parent.parent / ".coverage-threshold.json"
DEFAULT_REPORT = Path(__file__).parent.parent / ".coverage.json"


def load_coverage_report(report_path):
    """Load and validate coverage JSON report."""
    if not report_path.exists():
        print(f"ERROR: Coverage report not found: {report_path}")
        print("Run `make test-cov` first to generate the report.")
        sys.exit(2)
    return json.loads(report_path.read_text())


def calculate_module_coverage(files):
    """Calculate branch coverage percentage per module.

    Aggregates covered_branches / num_branches across all files in each module.
    """
    module_stats = {}

    for filepath, data in files.items():
        # Strip src/ prefix: src/trtutils/core/_memory.py -> trtutils/core/_memory.py
        rel = filepath.replace("src/", "")
        parts = rel.split("/")

        # Determine module key
        if len(parts) >= 3:
            # e.g., trtutils/core/_memory.py -> trtutils/core
            module = "/".join(parts[:2])
        elif len(parts) == 2:
            # e.g., trtutils/_engine.py -> trtutils/_engine
            module = parts[0] + "/" + Path(parts[1]).stem
        else:
            continue

        if module not in module_stats:
            module_stats[module] = {"covered": 0, "total": 0}

        summary = data["summary"]
        module_stats[module]["covered"] += summary.get("covered_branches", 0)
        module_stats[module]["total"] += summary.get("num_branches", 0)

    return {
        mod: (s["covered"] / s["total"] * 100 if s["total"] > 0 else 100.0)
        for mod, s in module_stats.items()
    }


def main():
    update_only = "--update" in sys.argv
    verbose = "--verbose" in sys.argv

    # Find report file
    report_path = DEFAULT_REPORT
    if "--report" in sys.argv:
        idx = sys.argv.index("--report")
        if idx + 1 < len(sys.argv):
            report_path = Path(sys.argv[idx + 1])

    # Load threshold file
    if not THRESHOLD_FILE.exists():
        print(f"ERROR: Threshold file not found: {THRESHOLD_FILE}")
        sys.exit(2)
    thresholds = json.loads(THRESHOLD_FILE.read_text())

    # Load and parse coverage report
    cov_data = load_coverage_report(report_path)
    coverage = calculate_module_coverage(cov_data.get("files", {}))

    if verbose:
        print("\n--- Per-Module Coverage ---")
        for mod, pct in sorted(coverage.items()):
            print(f"  {mod}: {pct:.1f}%")
        print()

    failed = False
    for module, threshold in sorted(thresholds.items()):
        current = coverage.get(module, 0.0)
        if current < threshold and not update_only:
            print(f"FAIL: {module} coverage {current:.1f}% < threshold {threshold:.1f}%")
            failed = True
        elif current > threshold:
            print(f"UP:   {module} {threshold:.1f}% -> {current:.1f}%")
            thresholds[module] = round(current, 1)
        else:
            print(f"OK:   {module} at {current:.1f}%")

    if not failed or update_only:
        THRESHOLD_FILE.write_text(json.dumps(thresholds, indent=4) + "\n")

    return 1 if failed else 0


if __name__ == "__main__":
    sys.exit(main())
```

**Key design:** This script is validate-only. It reads `.coverage.json` generated by `make test-cov`. It does NOT run pytest. This separation of concerns makes the pre-commit hook fast (just reads a file) and keeps test execution under the developer's control.

## Step 7: Create Root conftest.py with Core Shared Fixtures

**File:** `tests/conftest.py`

```python
"""Root conftest.py — shared fixtures for the trtutils test suite."""
from __future__ import annotations

from pathlib import Path
from typing import Callable

import numpy as np
import pytest

# ---------------------------------------------------------------------------
# Path constants
# ---------------------------------------------------------------------------
BASE_DIR = Path(__file__).parent.parent
DATA_DIR = BASE_DIR / "data"
IMAGES_DIR = DATA_DIR
ENGINES_DIR = DATA_DIR / "engines"

HORSE_IMAGE_PATH = IMAGES_DIR / "horse.jpg"
PEOPLE_IMAGE_PATH = IMAGES_DIR / "people.jpeg"
IMAGE_PATHS = [HORSE_IMAGE_PATH, PEOPLE_IMAGE_PATH]


# ---------------------------------------------------------------------------
# TRT version detection (for engine path versioning)
# ---------------------------------------------------------------------------
def get_trt_version() -> str:
    """Get TensorRT version string for engine path caching."""
    try:
        from trtutils.compat._libs import trt

        return trt.__version__
    except Exception:
        return "unknown"


TRT_VERSION = get_trt_version()


def version_engine_path(path: Path) -> Path:
    """Insert TRT version into engine filename for caching."""
    return path.with_name(f"{path.stem}_{TRT_VERSION}{path.suffix}")


# ---------------------------------------------------------------------------
# GPU availability
# ---------------------------------------------------------------------------
@pytest.fixture(scope="session")
def gpu_available() -> bool:
    """Check if a CUDA GPU is available."""
    try:
        from trtutils.compat._libs import cudart

        err, count = cudart.cudaGetDeviceCount()
        return count > 0
    except Exception:
        return False


@pytest.fixture(autouse=True)
def _skip_gpu_tests(request: pytest.FixtureRequest, gpu_available: bool) -> None:
    """Auto-skip tests marked @pytest.mark.gpu when no GPU is available."""
    if request.node.get_closest_marker("gpu") and not gpu_available:
        pytest.skip("No CUDA GPU available")


@pytest.fixture(autouse=True)
def _skip_dla_tests(request: pytest.FixtureRequest) -> None:
    """Auto-skip tests marked @pytest.mark.dla when not on Jetson."""
    if request.node.get_closest_marker("dla"):
        try:
            from trtutils._flags import FLAGS

            if not FLAGS.IS_JETSON:
                pytest.skip("DLA tests require Jetson hardware")
        except Exception:
            pytest.skip("Cannot determine Jetson status")


# ---------------------------------------------------------------------------
# Test images
# ---------------------------------------------------------------------------
@pytest.fixture(scope="session")
def test_images() -> list[np.ndarray]:
    """Load all test images as numpy arrays (BGR, uint8)."""
    import cv2

    images = []
    for path in IMAGE_PATHS:
        img = cv2.imread(str(path))
        if img is not None:
            images.append(img)
    return images


@pytest.fixture(scope="session")
def horse_image() -> np.ndarray:
    """Load the horse test image."""
    import cv2

    img = cv2.imread(str(HORSE_IMAGE_PATH))
    if img is None:
        pytest.skip("Horse test image not found")
    return img


# ---------------------------------------------------------------------------
# Parametrized fixtures
# ---------------------------------------------------------------------------
@pytest.fixture(
    params=[1, 2, 4],
    ids=["batch1", "batch2", "batch4"],
)
def batch_size(request: pytest.FixtureRequest) -> int:
    """Parametrized batch size for batch processing tests."""
    return request.param


# ---------------------------------------------------------------------------
# Factory fixtures
# ---------------------------------------------------------------------------
@pytest.fixture(scope="session")
def random_images() -> Callable[..., list[np.ndarray]]:
    """Factory fixture: generate random uint8 images.

    Usage:
        images = random_images(count=4, height=480, width=640)
    """
    rng = np.random.default_rng(42)

    def _make(
        count: int = 1,
        height: int = 480,
        width: int = 640,
        channels: int = 3,
    ) -> list[np.ndarray]:
        return [
            rng.integers(0, 255, (height, width, channels), dtype=np.uint8)
            for _ in range(count)
        ]

    return _make


@pytest.fixture(scope="session")
def build_test_engine() -> Callable[..., Path]:
    """Factory fixture: build and cache a TRT engine from an ONNX file.

    Engines are cached with TRT version in the filename so they
    are automatically rebuilt when TRT is upgraded.
    """

    def _build(
        onnx_path: Path,
        engine_dir: Path | None = None,
        **kwargs: object,
    ) -> Path:
        if engine_dir is None:
            engine_dir = ENGINES_DIR
        engine_dir.mkdir(parents=True, exist_ok=True)

        engine_path = version_engine_path(
            engine_dir / f"{onnx_path.stem}.engine"
        )

        if not engine_path.exists():
            from trtutils.builder import build_engine

            build_engine(
                onnx_path,
                engine_path,
                optimization_level=kwargs.pop("optimization_level", 1),
                **kwargs,
            )

        return engine_path

    return _build
```

### Key design choices:
- **Session-scoped GPU detection** — checked once, used by autouse skip fixture
- **Autouse skip fixtures** — tests marked `@pytest.mark.gpu` or `@pytest.mark.dla` auto-skip when hardware unavailable. No need for per-test `skipif`.
- **Version-aware engine caching** — preserves the existing pattern from `tests/paths.py`
- **Factory fixtures** — `random_images()` and `build_test_engine()` return callables for flexible use
- **Deterministic RNG** — `seed=42` for reproducible random test images
- **No mocking infrastructure** — per design decision, tests use real GPU

### 8a: Update ci/run_tests.sh

**File:** `ci/run_tests.sh`

The existing script runs `python3 -m pytest -rP -v tests/` which will discover both legacy and new tests after the move. Update it to ignore legacy:

```makefile
test:

	python3 -m pytest --cov=src/trtutils --cov-branch --cov-report=term-missing --cov-report=json:.coverage.json tests/ --ignore=tests/legacy/

```

**Note:** `test-cov` and `test-cov-html` both generate `.coverage.json` for the ratcheting script. The workflow is: `make test-cov` then `python ci/check_coverage.py`.

```yaml
      # Coverage threshold check

        name: Coverage threshold check
```
